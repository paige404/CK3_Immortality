grant_immortality_interaction = {
	category = interaction_category_uncategorized
	use_diplomatic_range = no
	ignores_pending_interaction_block = yes
	is_shown = {
		scope:recipient = {
			NOT = {

			}
		}
	}
	on_accept = {
		scope:recipient = {
			add_trait = immortality
		}
	}
	auto_accept = yes
	ai_will_do = {
		base = 0
	}
}
